Object subclass: Counter [
  |count| "Variable de instancia"



  class << new [
    ^super new initialize: 0.
  ]







  initialize: aValue [
    count := aValue.
    ^self.
  ]





  next [
    self initialize: count+1.
    ^count.
  ]



  nextIf: condition [
    
    ^condition ifTrue: [self next] 
               ifFalse: [count].
  ]
]



Counter subclass: FlexibleCounter [
  | block | "Variable de instancia"

  

  class << new: aBlock [
    ^super new useBlock: aBlock.
  ]





  useBlock: aBlock [
    block := aBlock.
    ^self.
  ]





  next [
  
    self initialize: (block value: count).
    ^count.
  ]
]
